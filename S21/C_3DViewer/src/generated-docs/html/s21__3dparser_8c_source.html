196 void s21_centring(Shape* shape) {
197  double minCoord[COORDS] = {shape->vertexes[0], shape->vertexes[1],
198  shape->vertexes[2]};
199  double maxCoord[COORDS] = {shape->vertexes[0], shape->vertexes[1],
200  shape->vertexes[2]};
201  // поиск мин и макс координат
202  for (unsigned i = 0; i < shape->countVertexes; ++i) {
203  if (shape->vertexes[i] < minCoord[i % 3])
204  minCoord[i % 3] = shape->vertexes[i];
205  if (shape->vertexes[i] > maxCoord[i % 3])
206  maxCoord[i % 3] = shape->vertexes[i];
207  }
208  // поиск центра и шкалы
209  double center[COORDS], minScale = 1, maxScale = 1;
210  for (int i = 0; i < 3; ++i) {
211  center[i] = (maxCoord[i] + minCoord[i]) / 2;
212  maxCoord[i] -= center[i];
213  minCoord[i] -= center[i];
214  minScale = minScale > fabs(maxCoord[i]) ? fabs(maxCoord[i]) : minScale;
215  minScale = minScale > fabs(minCoord[i]) ? fabs(minCoord[i]) : minScale;
216  maxScale = maxScale < fabs(maxCoord[i]) ? fabs(maxCoord[i]) : maxScale;
217  maxScale = maxScale < fabs(minCoord[i]) ? fabs(minCoord[i]) : maxScale;
218  }
219  // шкалирование
220  shape->scale = maxScale >= 1 || minScale < 1e-6 ? 1 / maxScale : 1 / minScale;
221  for (unsigned i = 0; i < shape->countVertexes; ++i)
222  shape->vertexes[i] =
223  (shape->vertexes[i] - center[i % 3]) * shape->scale * 0.8;
224  shape->scale = 1;
225 }
